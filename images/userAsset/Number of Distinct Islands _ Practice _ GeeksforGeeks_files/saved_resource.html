class Solution {
  public:
    
    void dfs(int row, int col, vector<vector<int>>& vis, vector<vector<int>>& grid, vector<pair<int,int>> &v, 
             int row0, int col0) {
        
        int n = grid.size();
        int m = grid[0].size();
        
        if(row<0 || row>=n || col<0 || col>=m || grid[row][col] == 0 || vis[row][col] == 1) {
            return ;
        }
        
        vis[row][col] = 1;
        v.push_back({row - row0, col - col0});
        
        dfs(row-1, col, vis, grid, v, row0, col0);
        dfs(row+1, col, vis, grid, v, row0, col0);
        dfs(row, col-1, vis, grid, v, row0, col0);
        dfs(row, col+1, vis, grid, v, row0, col0);
    }
    
    int countDistinctIslands(vector<vector<int>>& grid) {
        
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>> vis(n, vector<int>(m, 0));
        set<vector<pair<int,int>>> st;
        
        for(int i=0; i<n; i++) {
            for(int j=0; j<m; j++) {
                
                if(!vis[i][j] && grid[i][j] == 1) {
                    vector<pair<int,int>> v;
                    dfs(i, j, vis, grid, v, i, j);
                    st.insert(v);
                }
            }
        }
        return st.size();
        
        //T.C :- O(N * M) + O(N * M)*4  =>  O(N * M)

};

class Solution {
  public:
  
    void dfs(vector<vector<int>>& grid , vector<vector<bool>> &visited , vector<pair<int,int>> &vec , int i ,int j , int n , int m , int baseRow , int baseColumn){
        
        visited[i][j]=true;
        vec.push_back( {i-baseRow , j-baseColumn} );
        
        int delta_row[] = {-1 , 0 , 1 , 0 };
        int delta_col[] = { 0 , 1 , 0 , -1};
        
        for(int k=0 ; k<4 ; k++){
            int nrow = i+delta_row[k];
            int ncol = j+delta_col[k];
            if(nrow>=0 && nrow<n && ncol>=0 && ncol<m && !visited[nrow][ncol] && grid[nrow][ncol]==1){
                dfs(grid , visited , vec , nrow , ncol , n , m , baseRow , baseColumn);
            }
        }
    }
    
    
    int countDistinctIslands(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        
        vector<vector<bool>> visited(n,vector<bool>(m,false));
        
        set<vector<pair<int,int>>> s;
        
        for(int i=0 ; i<n ; i++){
            for(int j=0 ; j<m ; j++){
                if(grid[i][j]==1 && !visited[i][j]){
                    vector<pair<int,int>> vec;
                    dfs(grid , visited , vec ,i , j , n , m , i , j);
                    s.insert(vec);
                }
            }
        }
        
        return s.size();
        
    }
};

For individuals who are storing the path as a string, there is an important consideration they need to address. Let's consider an example:

Suppose we have the following matrix:

1 1 0 1 1
1 0 0 0 1
1 0 0 0 1

If someone is storing the path recursively using directions such as start (s), up (u), down (d), left (l), and right (r), then the path for both islands will become "srdd". However, upon closer inspection, it is evident that both islands are different. In such cases, after the recursion for a particular (i, j) is completed, introducing a random variable can help differentiate the paths.

By adding a random variable to the path, each distinct island will have a unique path representation,

1. traverse the grid find the unvisited islands and make all 1s visited 

2. for each island make a list and store {baseRow - row, baseCol - col} inside the list for every dfs call.

3.  then make this list a string and store it inside SET.

class Solution {

    int countDistinctIslands(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        int visited[][] = new int[n][m];
        
        HashSet<String> set = new HashSet<>();
        
        for (int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(visited[i][j] == 0 && grid[i][j] == 1){
                    ArrayList<int[]> list = new ArrayList<>();
                    dfs(grid,visited,i,j,list,i,j);
                    set.add(makeStr(list));
                }
                
            }
        }
        return set.size();
    }
    
    String makeStr(ArrayList<int[]> list){

        String ans = "";
        for (int[] val: list) {
            for (int num: val) {
                ans += num;
            }
        }

        return ans;
    }

        dfs(grid,visited,row,col+1,list,baseRow,baseCol);
        dfs(grid,visited,row,col-1,list,baseRow,baseCol);

class Solution {

        int i, j, n = ar.size(), m = ar[0].size();
        
        vis[rw][cl] = 1;
        
        rowColVec.push_back(rw  - parrow);
        rowColVec.push_back(cl  - parcol);
        
        for(int i = 0;i<4;i++){
            int row = rw + rdelta[i];
            int col = cl + cdelta[i];
            
            if(row >= 0 && row < n && col >= 0 && col < m && !vis[row][col] && ar[row][col] == 1){
                dfs(ar, vis, row, col, rowColVec, parrow, parcol);
            }
        }
    }
  public:

        // code here

        vector<vector<int>> vis(n, vector<int>(m, 0));

            for(j = 0;j<m;j++){

                    rowColVec.clear();

                   q.push({i,j});
                   vis[i][j] = 1;
                   while(!q.empty()){
                       int x = q.front().first;
                       int y = q.front().second;
                       temp.push_back({x-i,y-j});
                       q.pop();
                       vector<int> dx = {1,0,-1,0};
                       vector<int> dy = {0,1,0,-1};
                       for(int i=0;i<4;i++){
                           int nx = x+dx[i];
                           int ny = y+dy[i];
                           if(nx>=0 && nx<n && ny>=0 && ny<m && !vis[nx][ny] && grid[nx][ny]==1){
                               q.push({nx,ny});
                               vis[nx][ny] = 1;
                           }
                       }
                       
                   }
                   s.insert(temp);
               }
           }
        }
        return s.size();
    }
};
// EASY DFS SOLUTION IN C++ O(N*M),O(N*M)
class Solution {
  public:
  void dfs(int bx,int by,int x,int y,vector<vector<int>>&vis,
  vector<vector<int>>&grid,vector<pair<int,int>>&temp,int n,int m){
      vis[x][y] = 1;
      temp.push_back({x-bx,y-by});
      vector<int> dx = {1,0,-1,0};
      vector<int> dy = {0,1,0,-1};
      for(int i=0;i<4;i++){
          int nx = x+dx[i];
          int ny = y+dy[i];
          if(nx>=0 && nx<n && ny>=0 && ny<m && !vis[nx][ny] && grid[nx][ny]==1)
          dfs(bx,by,nx,ny,vis,grid,temp,n,m);
      }
      
  }
    int countDistinctIslands(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>>vis(n,vector<int>(m,0));
        set<vector<pair<int,int>>>s;
        for(int i=0;i<n;i++)
           for(int j=0;j<m;j++)
               if(grid[i][j]==1 && !vis[i][j]){
                   vector<pair<int,int>> temp;
                   dfs(i,j,i,j,vis,grid,temp,n,m);
                   s.insert(temp);
               }
        return s.size();
    }
};
